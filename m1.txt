#include <iostream>
using namespace std;

struct Node {
	float coef;
	int expo;
	Node *next;
};

struct SingleLinkedList {
	Node *begin;
	Node *end;
};

bool isEmpty(SingleLinkedList &l) {
	return (l.begin == NULL) ? true : false;
};

void freeMemory(SingleLinkedList &l) {
    Node *aux = l.begin;
    Node *n;
    while (aux != NULL) {
        n = aux;
        aux = aux->next;
        delete n;
    }
};

// ======== TAD linked List =============

void initialize(SingleLinkedList &l) {
	l.begin = NULL;
	l.end = NULL;
};

// order polinomio per exponent
bool insert(SingleLinkedList &l, float coef, int expo) {
	Node *n = new Node;
	if ( n == NULL ) return false;
	n->coef = coef;
	n->expo = expo;
	n->next = NULL;

	if (l.begin == NULL) { // insert into an empty list
		l.begin = n;
		l.end = n;
	}
	else if (expo < l.begin->expo) { // insert at the beggining
		n->next = l.begin;
		l.begin = n;
	}
	else if (expo > l.end->expo) { // insert at the ending
		l.end->next = n;
		l.end = n;
	}
	else { // insert in the middle
		Node *actual = l.begin;
		while (actual->next != NULL) {
			Node *next = actual->next;
			if (actual->expo < expo && expo < next->expo) {
				actual->next = n;
				n->next = next;
			}
			if (actual->coef == coef) { // it not working this logic
			    n->coef += actual->coef;
			    actual = n;
			}
			actual = actual->next;
		}
	}
	return true;
};

bool removeMono(SingleLinkedList &l, int expo) {
    if (isEmpty(l)) {
        cout << "No monomy was removed!" << endl;
        return false;
    }
    
    // removing at the beggining
    if (l.begin->expo == expo) {
        Node *aux = l.begin;
        if (l.end->expo == expo) { // just one nomomy on the list
            l.begin = NULL;
            l.end = NULL;
            delete aux;
            return true;
        }
        l.begin = l.begin->next;
        delete aux;
        return true;
    }
    else {
        Node *before = l.begin;
        while (before != NULL) {
            if (before->next != NULL && before->next->expo == expo) {
                Node *nodeToRemove = before->next;
                if (l.end == nodeToRemove) { // remove at the ending
                    before->next = NULL;
                    l.end = before;
                    delete nodeToRemove;
                    return true;
                }
                // remove at the middle
                before->next = nodeToRemove->next;
                delete nodeToRemove;
                return true;
            }
            before = before->next;
        }
    }
    return true;
};

bool search(SingleLinkedList &l, float coef) {
    Node *aux = l.begin;
    while (aux != NULL) {
        if (aux->coef == coef) return true;
        aux = aux->next;
    }
    return false;
}

void show(SingleLinkedList &l) {
	Node *aux = l.begin;
	cout << "P = ";
	while (aux != NULL) {
	    if (aux == l.end) {
	        cout << aux->coef << "x^" << aux->expo;
	        break;
	    }
		cout << aux->coef << "x^" << aux->expo << " + ";
		aux = aux->next;
	};
};

// ======== Polinomial Operations =============

/* operate()
   receive two polinomials (both are SingleLinkedList);
   operation -> signal of operation which the function will be performing
      '+' -> sum up two monomials
      '-' -> subtract two monomials
*/
SingleLinkedList operate(SingleLinkedList &poli_one, SingleLinkedList &poli_two, char operation) {
    if (isEmpty(poli_one) && isEmpty(poli_two)) {
        return SingleLinkedList{};
    }
    
    SingleLinkedList new_poli;
    initialize(new_poli);
    
    Node *mono_one = poli_one.begin;
    while (mono_one != NULL) {
        Node *mono_two = poli_two.begin;
        while(mono_two != NULL) {
            if (mono_one->expo == mono_two->expo) {
                if (operation == '+') {
                    insert(new_poli, (mono_one->coef + mono_two->coef), mono_one->expo);
                } else {
                    insert(new_poli, (mono_one->coef - mono_two->coef), mono_one->expo);
                }
            }
            mono_two = mono_two->next;
        }
        mono_one = mono_one->next;
    }
    return new_poli;
}
