#include <iostream>
using namespace std;

template <typename T>
struct Node {
	T element;
	Node<T> *next;
};

// LUE (Lista unicamente encadeade)
template <typename T>
struct SingleLinkedList {
	Node<T> *begin;
	Node<T> *end;
};

template <typename T>
void initialize(SingleLinkedList<T> &l) {
	l.begin = NULL;
	l.end = NULL;
};

template <typename T>
bool isEmpty(SingleLinkedList<T> &l) {
	return (l.begin == NULL) ? true : false;
}

template <typename T>
int length(SingleLinkedList<T> &l) {
	Node<T> *aux = l.begin;
	int count = 0;
	while (aux != NULL) {
		count++;
		aux = aux->next;
	};
	cout << "Length: " << count << endl;
	return count;
};

template <typename T>
void show(SingleLinkedList<T> &l) {
	Node<T> *aux = l.begin;
	while (aux != NULL) {
		cout << aux->element << " ";
		aux = aux->next;
	};
};

template <typename T>
bool search(SingleLinkedList<T> &l, T element) {
    Node<T> *aux = l.begin;
    while (aux != NULL) {
        if (aux->element == element) return true;
        aux = aux->next;
    }
    return false;
};

template <typename T>
bool insert(SingleLinkedList<T> &l, T element) {
	Node<T> *n = new Node<T>;
	if ( n == NULL ) return false;
	n->element = element;
	n->next = NULL;
	//initializeNode(n, valor);

	if (l.begin == NULL) { // INSERT EMPTY LIST
		l.begin = n;
		l.end = n;
		cout << "\nSuccessful insert" << endl;
	}
	else if (element < l.begin->element) { // INSERT TO THE BEGGINING
		n->next = l.begin;
		l.begin = n;
	}
	else if (element > l.end->element) { // INSERT TO THE END
		l.end->next = n;
		l.end = n;
		cout << "\nSuccessful insert" << endl;
	}
	else { // INSERT IN THE MIDDLE
		Node<T> *aux = l.begin;
		while (aux->next != NULL) {
			Node<T> *next = aux->next;
			if (aux->element < element && element < next->element) {
				aux->next = n;
				n->next = next;
			}
			aux = aux->next;
		}
	}

	return true;
};

template <typename T>
bool removeFrom(SingleLinkedList<T> &l, T element) {
	if (isEmpty(l)) { // remove from empty list
		cout << "\nNo element was removed!" << endl;
		return false;
	}

	if (l.begin->element == element) { // remove from the beggining
		Node<T> *aux = l.begin;
		if (l.begin == l.end) { // just one element on the list
			l.begin = NULL;
			l.end = NULL;
			delete aux;
			cout << "Item: " << element << " removed!" << endl;
			return true;
		}
		l.begin = l.begin->next;
		delete aux;
		cout << "Item: " << element << " removed!" << endl;
		return true;
	}
	else {
		Node<T> *bef = l.begin;
		while (bef != NULL) {
			if (bef->next != NULL && bef->next->element == element) {

				Node<T> *nodeToRemove = bef->next;
				if (l.end == nodeToRemove) { // remove from the ending
					bef->next = NULL;
					l.end = bef;
					delete nodeToRemove;
					cout << "Item: " << element << " removed!" << endl;
					return true;
				}

				// remove from the middle
				bef->next = nodeToRemove->next;
				delete nodeToRemove;
				cout << "Item: " << element << " removed!" << endl;
				return true;
			}
			bef = bef->next;
		}
	}

	cout << "No element was find for: " << element << endl;
	return false;
};

template <typename T>
bool equal(SingleLinkedList<T> &l1, SingleLinkedList<T> &l2) {
    if (length(l1) != length(l2)) {
        cout << "Both lists are not the same size!" << endl;
        return false;
    }
    
    Node<T> *aux1 = l1.begin;
    while (aux1 != NULL) {
        cout << "List 1. Checking: " << aux1->element << endl;
        Node<T> *aux2 = l2.begin;
        while (aux2 != NULL) {
            cout << "List 2. Checking: " << aux2->element << endl;
            if (aux1->element == aux2->element) {
                aux2 = aux2->next;
                break;
            }
        }
        aux1 = aux1->next;
    };
    cout << "l1 and l2 are equal!" << endl;
    return true;
};

template <typename T>
void copy(SingleLinkedList<T> &l1, SingleLinkedList<T> &l2) {
    if (isEmpty(l1)) { 
        cout << "List 1 is empty! No value to be copied." << endl;
        return;
    }
    
    Node<T> *aux = l1.begin;
    while (aux != NULL) {
        insert(l2, aux->element);
        aux = aux->next;
    };
    
    cout << "All values from list 1 were copied to list 2!" << endl;
};

template <typename T>
SingleLinkedList<T> diference(SingleLinkedList<T> &l1, SingleLinkedList<T> &l2) {
    if (isEmpty(l1) && isEmpty(l2)) { 
        cout << "Both lists are empty!" << endl;
        return SingleLinkedList<int> {};
    }
    
    SingleLinkedList<int> l3;
    initialize(l3);
    
    Node<T> *aux = l1.begin;
    while (aux != NULL) {
        if (!search(l2, aux->element)) {
            insert(l3, aux->element);
        }
        aux = aux->next;
    };
    
    return l3;
};
